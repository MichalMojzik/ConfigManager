36  bool Configuration::ResolveLink(const std::string& value, LinkValues& link_values, std::string* result, std::vector<OptionNode*>* resolved_links, Link* unresolved_link)
37  {
38  std::size_t offset = 0;
39  while(offset < value.size())
40  {
41  // nalezeni dalsiho linku
42  auto link_position = find_first_nonespaced(value, '$', offset);
43  if(link_position == std::string::npos)
44  {
45  // zadny link nenalezen, vypsat zbytek vstupu a skoncit
46  if(result != nullptr)
47  {
48  *result += value.substr(offset);
49  }
50  break;
51  }
52 
53  // vypiseme vse az do linku
54  if(result != nullptr)
55  {
56  *result += value.substr(offset, link_position - offset);
57  }
58 
59  // pokud retezec dal nepokracuje, nebo nepokracuje otevrenou slozenou zavorkou, je problem se vstupem
60  if(link_position + 1 == value.length() || value[link_position + 1] != '{')
61  {
62  throw MalformedInputException();
63  }
64 
65  // najdeme oddelovac jmen sekce a volby, ktery je v aktulnim urovni zanoreni
66  auto hash_position = find_first_nonespaced_unscoped(value, '#', link_position + 2, '{', '}');
67  if(hash_position == std::string::npos) // pokud neni, chyba
68  {
69  throw MalformedInputException();
70  }
71  // najdeme koncovou zavorku, ktera je v aktulnim urovni zanoreni
72  auto last_brace_position = find_first_nonespaced_unscoped(value, '}', hash_position + 1, '{', '}');
73  if(last_brace_position == std::string::npos) // pokud neni, chyba
74  {
75  throw MalformedInputException();
76  }
77 
78  // par jmen {sekce, volba}
79  Link link;
80  // retezec oznacujici jmeno sekce muze obsahovat linky, pokusime se je rekurzivne vyhodnotit
81  if(!ResolveLink(value.substr(link_position + 2, hash_position - link_position - 2), link_values, &link.first, resolved_links, unresolved_link))
82  {
83  // vyhodnoceni linku se nezdarilo, koncime; nevyhodnoceny link je prirazen do unresolved_link z rekurzivniho volani
84  return false;
85  }
86  // retezec oznacujici jmeno volby muze obsahovat linky, pokusime se je rekurzivne vyhodnotit
87  if(!ResolveLink(value.substr(hash_position + 1, last_brace_position - hash_position - 1), link_values, &link.second, resolved_links, unresolved_link))
88  {
89  // vyhodnoceni linku se nezdarilo, koncime; nevyhodnoceny link je prirazen do unresolved_link z rekurzivniho volani
90  return false;
91  }
92 
93  // pokusime se vyhodnotit link urceny ziskanymi jmeny
94  auto link_it = link_values.find(link);
95  if(link_it == link_values.end())
96  {
97  // link nenalezen, ohlasime ho
98  if(unresolved_link != nullptr)
99  {
100  *unresolved_link = link;
101  }
102  // a selzeme
103  return false;
104  }
105 
106  // link lze vyresit, zapiseme ho tedy do seznamu
107  if(resolved_links)
108  {
109  resolved_links->push_back(link_it->second);
110  }
111 
112  // hodnotu linku ziskame z OptionNode ziskaneho pres link
113  if(result != nullptr)
114  {
115  *result += link_it->second->Value();
116  }
117 
118  // skocime za konec specifikace linku a opakujeme
119  offset = last_brace_position + 1;
120  }
121 
122  return true;

192  void Configuration::ProcessSection(SectionRange range, LinkValues& link_values, PostponedSections& postponed_sections, PostponedOptions& postponed_options)
193  {
194  // ziskame prvni radek ze zadaneho rozsahu bez komentaru a prebytecnych mezer
195  std::string line = StripComment(original_lines_[range.first].line_);
196  if(line.front() != '[' || line.back() != ']')
197  {
198  throw InvalidOperationException();
199  }
200 
201  // nazev sekce muze obsahovat linky, pokusime se je vyhodnotit
202  std::vector<OptionNode*> dependancies;
203  Link unresolved_link;
204  std::string section_name;
205  if(!ResolveLink(line.substr(1, line.length() - 2), link_values, &section_name, &dependancies, &unresolved_link))
206  {
207  // nektere linky nebylo mozne vyhodnotit, ulozime si tuto sekci pro zpracovani na dobu, kdy problematicky link jiz vyhodnotit pujde
208  postponed_sections.emplace(unresolved_link, range);
209  return;
210  }
211  section_name = unescape(section_name);
212 
213  // vytvorime novou sekci, ci ziskame tu, ktera jiz byla vytvorena driv (v ramci nacitani nebo v ramci pristupu k sekci uzivatelem)
214  auto& section = RetrieveSection(section_name);
215  // ulozime si linky, na kterych je jmeno zavisle, abychom mohli pote pri ukladani provest kontrolu, zda muzeme zachovat puvodni radek
216  section.name_dependant_on_ = std::move(dependancies);
217  // oznacime sekci za nactenou (muze vyhodit vyjimku, pokud byla sekce jiz jednou nactena)
218  section.Load();
219 
220  // oznacime uvodni radek sekce tak, ze odpovida aktualni sekci (pro ucely rekonstrukce pri ukladani)
221  original_lines_[range.first].section_ = &section;
222  for(std::size_t line_index = range.first + 1; line_index <= range.second; ++line_index)
223  {
224  // kazdy radek ze stanoveneho rozsahu oznacime tak, ze odpovida aktualni sekci
225  original_lines_[line_index].section_ = &section;
226

231  void Configuration::ProcessOption(OptionIndex index, LinkValues& link_values, PostponedSections& postponed_sections, PostponedOptions& postponed_options)
232  {
233  // vyextrahujeme data radky jakozto referenci
234  auto& original_line_data = original_lines_[index];
235 
236  // ziskame radek bez komentaru a prebytecnych mezer
237  std::string line = StripComment(original_line_data.line_);
238  // prazdny radek neresime
239  if(line.empty())
240  {
241  return;
242  }
243 
244  // najdeme rovnitko
245  auto assignment_position = find_first_nonespaced(line, '=');
246  // pokud rovnitko neexistuje, pak se nejedna o volbu, i kdyz jsme ji ocekavali, tedy se nejedna o platny .ini format
247  if(assignment_position == std::string::npos)
248  {
249  throw MalformedInputException();
250  }
251 
252  Link unresolved_link;
253  // pokusime se ziskat jiz drive prirazenou volbu k tomuto radku
254  OptionNode* option = original_line_data.option_;
255  // pokud zatim zacna volba prirazena neni, musime zpracovat nazev volby a vytvorit ji
256  if(option == nullptr)
257  {
258  // nazev volby muze obsahovat linky, pokusime se je vyhodnotit
259  std::vector<OptionNode*> dependancies;
260  std::string option_name;
261  if(!ResolveLink(trim_nonescaped(line.substr(0, assignment_position)), link_values, &option_name, &dependancies, &unresolved_link))
262  {
263  // nektere linky nebylo mozne vyhodnotit, ulozime si tuto volby pro zpracovani na dobu, kdy problematicky link jiz vyhodnotit pujde
264  postponed_options.emplace(unresolved_link, index);
265  return;
266  }
267  option_name = unescape(option_name);
268 
269  // vytvorime novou volbu, ci ziskame tu, ktera jiz byla vytvorena driv (v ramci nacitani nebo v ramci pristupu k sekci uzivatelem)
270  option = &(*original_line_data.section_)[option_name];
271  // ulozime si linky, na kterych je jmeno zavisle, abychom mohli pote pri ukladani provest kontrolu, zda muzeme zachovat puvodni radek
272  option->name_dependant_on_ = std::move(dependancies);
273  // poznacime oznacime radek tak, ze odpovida vytvorene volbe (pro ucely rekonstrukce pri ukladani nebo opetovneho zpracovani pri chybe linku)
274  original_line_data.option_ = option;
275  }
276 
277  // ziskame hodnotu volby, prozatim escapovanou a potencialne obsahujici linky
278  std::string raw_value = trim_nonescaped(line.substr(assignment_position + 1));
279  // hodnota volby muze obsahovat linky, pokusime se je vyhodnotit
280  std::string option_value;
281  if(!ResolveLink(raw_value, link_values, &option_value, &option->value_dependant_on_, &unresolved_link))
282  {
283  // nektere linky nebylo mozne vyhodnotit, ulozime si tuto volby pro zpracovani na dobu, kdy problematicky link jiz vyhodnotit pujde
284  postponed_options.emplace(unresolved_link, index);
285  // vycistime vsechny vyhodnocene linky, ktere se mohly ulozit
286  option->value_dependant_on_.clear();
287  return;
288  }
289  // oznacime rozsah radku, ve kterem je zapsana hodnota (pro ucely rekonstrukce pri ukladani)
290  original_line_data.value_start_ = line.find_first_not_of(' ', assignment_position + 1);
291  original_line_data.value_end_ = original_line_data.value_start_ + raw_value.length();
292 
293  // nactenou hodnotu ulozime do volby (muze vyhodit vyjimku, pokud byla volba jiz jednou nactena)
294  option->Load(option_value);
295 
296  // ulozime si volbu jakozto hodnotu linku a pokusime se zpracovat vsechny sekce a volby, ktere nebylo mozne zpracovat predtim
297  Link link = { option->Section().Name(), option->Name() };
298  link_values.emplace(link, option);
299  for(auto it_range = postponed_sections.equal_range(link); it_range.first != it_range.second; ++it_range.first)
300  {
301  ProcessSection(it_range.first->second, link_values, postponed_sections, postponed_options);
302  }
303  for(auto it_range = postponed_options.equal_range(link); it_range.first != it_range.second; ++it_range.first)
304  {
305  ProcessOption(it_range.first->second, link_values, postponed_sections, postponed_options);
306  }
